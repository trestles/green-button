def validate_string(arg, arg_name, required = true)
  return if !required && !arg
  validate_required(arg, arg_name)
  raise ArgumentError, "Argument #{arg_name} should be string." unless arg.respond_to?(:to_s) && !(/\S/ !~ arg.to_s)
end
